---
title: 'Create Token Sale'
api: 'POST /api/v1/investment/token-sale'
---

Deploy a new company token and create a token sale for capital raising.

## Request Body

<ParamField body="companyName" type="string" required>
  Name of the company raising capital (max 100 characters)
</ParamField>

<ParamField body="companyDescription" type="string" required>
  Description of the company and offering (max 5000 characters)
</ParamField>

<ParamField body="sector" type="string" required>
  Business sector or industry (max 100 characters)
</ParamField>

<ParamField body="targetRaise" type="number" required>
  Target amount to raise (in EUR, minimum 1000)
</ParamField>

<ParamField body="tokenSupply" type="number" required>
  Total number of tokens to create
</ParamField>

<ParamField body="companyPercentage" type="number" required>
  Percentage of company ownership represented by tokens (0.01-100)
</ParamField>

<ParamField body="paymentMethod" type="string" default="both">
  Accepted payment methods: "crypto", "bank", or "both"
</ParamField>

<ParamField body="fixedInvestment" type="boolean" default="false">
  Whether to enforce fixed investment amounts
</ParamField>

<ParamField body="minInvestment" type="number" optional>
  Minimum investment amount per investor (overrides framework default)
</ParamField>

<ParamField body="maxInvestment" type="number" optional>
  Maximum investment amount per investor (overrides framework default)
</ParamField>

<ParamField body="isPublic" type="boolean" default="true">
  Whether the sale is open to all investors or whitelist only
</ParamField>

<ParamField body="whitelist" type="array" optional>
  Array of whitelisted Ethereum addresses for private sales
</ParamField>

<ParamField body="futureMint" type="boolean" default="false">
  Whether tokens should be minted at a future date
</ParamField>

<ParamField body="mintDate" type="string" optional>
  ISO date string for future minting (required if futureMint is true)
</ParamField>

<ParamField body="issuerAddress" type="string" optional>
  Ethereum address of the token issuer
</ParamField>

<ParamField body="additionalData" type="object" optional>
  Additional metadata for the token sale
</ParamField>
