---
title: 'Sync Reserve Data'
api: 'POST /api/stablecoin/sync-reserves'
---

Synchronize off-chain reserve data for transparency and audit trail.

<ParamField body="stablecoinAddress" type="string" required>
  Address of the stablecoin contract
</ParamField>

<ParamField body="reserves" type="object" required>
  Reserve data object
  <Expandable title="reserves object">
    <ParamField body="totalReserves" type="string">
      Total reserves amount
    </ParamField>
    <ParamField body="currency" type="string">
      Reserve currency (e.g., "USD", "EUR")
    </ParamField>
    <ParamField body="bankAccount" type="string">
      Bank account identifier
    </ParamField>
    <ParamField body="custodian" type="string">
      Custodian name
    </ParamField>
    <ParamField body="lastAuditDate" type="string">
      ISO date of last audit
    </ParamField>
    <ParamField body="attestationUrl" type="string">
      URL to attestation document
    </ParamField>
    <ParamField body="metadata" type="object">
      Additional metadata
    </ParamField>
  </Expandable>
</ParamField>
