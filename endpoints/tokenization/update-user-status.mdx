---
title: 'Update User Status'
api: 'POST /api/v1/tokenization/user-status'
---

Update compliance status for multiple users including whitelist, KYC, freeze, and blacklist flags.

<ParamField body="tokenAddress" type="string" required>
  ERC20 token contract address
</ParamField>

<ParamField body="updates" type="array" required>
  List of user status updates
  
  <Expandable title="updates">
    <ParamField body="address" type="string" required>
      User Ethereum address
    </ParamField>
    
    <ParamField body="whitelisted" type="boolean">
      Set whitelist status
    </ParamField>
    
    <ParamField body="kyc" type="boolean">
      Set KYC verification status
    </ParamField>
    
    <ParamField body="frozen" type="boolean">
      Set account freeze status
    </ParamField>
    
    <ParamField body="blacklisted" type="boolean">
      Set blacklist status
    </ParamField>
  </Expandable>
</ParamField>
