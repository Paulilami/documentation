---
title: 'Create Token'
api: 'POST /api/v1/tokenization/create'
---

Deploy a new ERC20 token with configurable features and initial distribution.

<ParamField body="name" type="string" required>
  Token name (3-32 characters)
</ParamField>

<ParamField body="symbol" type="string" required>
  Token symbol (2-8 characters)
</ParamField>

<ParamField body="initialSupply" type="string" required>
  Initial token supply (will be adjusted for decimals)
</ParamField>

<ParamField body="decimals" type="number" default="18">
  Number of decimal places (0-18)
</ParamField>

<ParamField body="assetType" type="string" required>
  Asset classification type
  
  Allowed values: `EQUITY`, `BOND`, `FUND`, `REAL_ESTATE`, `CASH_EQUIVALENT`, `COMMODITY`, `DEBT_INSTRUMENT`, `ART`, `COLLECTIBLE`, `MUSIC_RIGHTS`, `FILM_RIGHTS`, `PATENT`, `TRADEMARK`, `LICENSE_CONTRACT`, `DATA_ASSET`, `CARBON_CREDIT`, `GAME_ITEM`, `DOMAIN_NAME`, `OTHER`
</ParamField>

<ParamField body="maxSupply" type="string">
  Maximum token supply cap
</ParamField>

<ParamField body="maxPerWallet" type="string">
  Maximum tokens per wallet
</ParamField>

<ParamField body="monthlyLimit" type="string">
  Monthly transfer limit per wallet
</ParamField>

<ParamField body="authority" type="string">
  Authority address with special permissions (Ethereum address)
</ParamField>

<ParamField body="feeBps" type="number" default="0">
  Transaction fee in basis points (0-10000)
</ParamField>

<ParamField body="feeReceiver" type="string">
  Address to receive transaction fees (Ethereum address)
</ParamField>

<ParamField body="headerImage" type="string">
  URL to token header image
</ParamField>

<ParamField body="recipients" type="array">
  Initial token distribution
  
  <Expandable title="recipients">
    <ParamField body="address" type="string" required>
      Recipient Ethereum address
    </ParamField>
    
    <ParamField body="amount" type="string" required>
      Token amount to allocate
    </ParamField>
  </Expandable>
</ParamField>

<ParamField body="features" type="object">
  Token feature configuration
  
  <Expandable title="features">
    <ParamField body="mintable" type="boolean" default="true">
      Enable minting after deployment
    </ParamField>
    
    <ParamField body="burnable" type="boolean" default="true">
      Allow token burning
    </ParamField>
    
    <ParamField body="kyc" type="boolean" default="false">
      Require KYC verification
    </ParamField>
    
    <ParamField body="blacklist" type="boolean" default="false">
      Enable blacklist functionality
    </ParamField>
    
    <ParamField body="whitelist" type="boolean" default="false">
      Enable whitelist functionality
    </ParamField>
    
    <ParamField body="freeze" type="boolean" default="false">
      Enable account freezing
    </ParamField>
    
    <ParamField body="docs" type="boolean" default="true">
      Enable documentation features
    </ParamField>
    
    <ParamField body="tags" type="boolean" default="true">
      Enable tagging features
    </ParamField>
    
    <ParamField body="mintFee" type="boolean" default="false">
      Charge fee for minting
    </ParamField>
    
    <ParamField body="burnFee" type="boolean" default="false">
      Charge fee for burning
    </ParamField>
    
    <ParamField body="txFee" type="boolean" default="false">
      Enable transaction fees
    </ParamField>
    
    <ParamField body="authority" type="boolean" default="false">
      Enable authority control
    </ParamField>
    
    <ParamField body="unlimitedSupply" type="boolean" default="false">
      Allow unlimited token supply
    </ParamField>
  </Expandable>
</ParamField>
