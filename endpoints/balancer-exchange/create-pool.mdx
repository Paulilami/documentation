---
title: 'Create Pool'
api: 'POST /api/balancer/v1/pools/create'
---

Create a new weighted pool for trading tokenized stocks against stablecoins.

<ParamField body="poolName" type="string" required>
  Name for the pool (e.g., "AAPL/USDC 80/20")
</ParamField>

<ParamField body="poolSymbol" type="string" required>
  Symbol for the pool token (e.g., "AAPL-USDC-BPT")
</ParamField>

<ParamField body="stockToken" type="string" required>
  Contract address of the tokenized stock
</ParamField>

<ParamField body="pairedToken" type="string" required>
  Contract address of the paired token (usually USDC)
</ParamField>

<ParamField body="stockISIN" type="string" required>
  International Securities Identification Number
</ParamField>

<ParamField body="stockWeight" type="number" default="80">
  Weight percentage for stock token (0-100)
</ParamField>

<ParamField body="pairedWeight" type="number" default="20">
  Weight percentage for paired token (0-100)
</ParamField>

<ParamField body="swapFeePercentage" type="number" default="0.003">
  Swap fee percentage (0.001 = 0.1%)
</ParamField>
