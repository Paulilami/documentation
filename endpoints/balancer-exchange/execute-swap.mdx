---
title: 'Execute Swap'
api: 'POST /api/balancer/v1/swap/execute'
---

Execute a token swap through a Balancer pool.

<ParamField body="poolId" type="string" required>
  The unique identifier of the pool
</ParamField>

<ParamField body="tokenIn" type="string" required>
  Address of the token being sold
</ParamField>

<ParamField body="tokenOut" type="string" required>
  Address of the token being bought
</ParamField>

<ParamField body="amountIn" type="string" required>
  Amount of tokens to sell
</ParamField>

<ParamField body="minAmountOut" type="string" default="0">
  Minimum amount of tokens to receive (slippage protection)
</ParamField>

<ParamField body="deadline" type="number">
  Unix timestamp for transaction deadline
</ParamField>
