---
title: 'Open Leverage Position'
api: 'POST /api/balancer/v1/leverage/position/open'
---

Open a leveraged trading position on a stock token.

<ParamField body="poolId" type="string" required>
  The unique identifier of the pool
</ParamField>

<ParamField body="stockISIN" type="string" required>
  International Securities Identification Number
</ParamField>

<ParamField body="stockToken" type="string" required>
  Contract address of the stock token
</ParamField>

<ParamField body="collateralAmount" type="string" required>
  Amount of collateral to deposit
</ParamField>

<ParamField body="leverage" type="number" required>
  Leverage multiplier (1-100x)
</ParamField>

<ParamField body="isLong" type="boolean" required>
  True for long position, false for short
</ParamField>

<ParamField body="stopLoss" type="string">
  Stop loss price (0 for none)
</ParamField>

<ParamField body="takeProfit" type="string">
  Take profit price (0 for none)
</ParamField>
