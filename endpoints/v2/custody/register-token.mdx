---
title: 'Register Token'
api: 'POST /api/v2/custody/api/tokens'
---

Register a token for custodial balance tracking. This creates a token entry in your instance that can be used to track customer balances.

## Request Body

<ParamField body="tokenAddress" type="string" required>
  On-chain token contract address
</ParamField>

<ParamField body="name" type="string" required>
  Human-readable token name
</ParamField>

<ParamField body="symbol" type="string" required>
  Token symbol (e.g., "GLD", "TSLA")
</ParamField>

<ParamField body="decimals" type="number" default="18">
  Token decimal places
</ParamField>

<ParamField body="metadataURI" type="string">
  URI pointing to token metadata JSON
</ParamField>

<ParamField body="assetType" type="string">
  Asset category (e.g., "commodity", "stock", "bond")
</ParamField>

## Response Fields

<ResponseField name="id" type="string">Unique token registry ID</ResponseField>
<ResponseField name="instanceId" type="string">Your instance ID</ResponseField>
<ResponseField name="tokenAddress" type="string">Normalized token address</ResponseField>
<ResponseField name="name" type="string">Token name</ResponseField>
<ResponseField name="symbol" type="string">Token symbol</ResponseField>
<ResponseField name="decimals" type="number">Decimal places</ResponseField>
<ResponseField name="metadataURI" type="string">Metadata URI</ResponseField>
<ResponseField name="assetType" type="string">Asset type</ResponseField>
<ResponseField name="isActive" type="boolean">Whether token is active</ResponseField>
<ResponseField name="createdAt" type="string">Creation timestamp</ResponseField>

    },

)
