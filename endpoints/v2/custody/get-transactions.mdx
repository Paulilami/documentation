---
title: 'Get Transactions'
api: 'GET /api/v2/custody/api/transactions'
---

Query the transaction ledger with optional filters. All balance operations create immutable transaction records for audit trails.

## Query Parameters

<ParamField query="customerRef" type="string">
  Filter by customer (returns transactions where customer is sender or recipient)
</ParamField>

<ParamField query="tokenAddress" type="string">
  Filter by token address
</ParamField>

<ParamField query="type" type="string">
  Filter by transaction type: `credit`, `debit`, `transfer`, `lock`, `unlock`
</ParamField>

<ParamField query="from" type="string">
  Start date filter (ISO 8601 format)
</ParamField>

<ParamField query="to" type="string">
  End date filter (ISO 8601 format)
</ParamField>

<ParamField query="limit" type="number" default="100">
  Maximum results (max 500)
</ParamField>

<ParamField query="offset" type="number" default="0">
  Results to skip for pagination
</ParamField>

## Response Fields

<ResponseField name="transactions" type="array">Array of transaction records</ResponseField>
<ResponseField name="total" type="number">Total matching transactions</ResponseField>

```javascript JavaScript
const params = new URLSearchParams({
  customerRef: 'CUST-12345',
  type: 'transfer',
  limit: '50'
});

const response = await fetch(
  `https://api.trusset.org/api/v2/custody/api/transactions?${params}`,
  {
    headers: {
      'x-api-key': 'trusset_v2_abc123xy_a1b2c3d4e5f6...'
    }
  }
);

const data = await response.json();
console.log(`Found ${data.data.total} transactions`);
```
```python Python
import requests

response = requests.get(
    'https://api.trusset.org/api/v2/custody/api/transactions',
    headers={
        'x-api-key': 'trusset_v2_abc123xy_a1b2c3d4e5f6...'
    },
    params={
        'customerRef': 'CUST-12345',
        'from': '2024-01-01T00:00:00Z',
        'to': '2024-01-31T23:59:59Z'
    }
)

data = response.json()
print(f"Found {data['data']['total']} transactions")
```

    transactions: Array<{
      id: string;
      type: 'credit' | 'debit' | 'transfer' | 'lock' | 'unlock';
      tokenAddress: string;
      token: { name: string; symbol: string } | null;
      fromCustomerRef: string | null;
      toCustomerRef: string | null;
      amount: string;
      reference: string | null;
      metadata: object | null;
      createdAt: string;
    }>;
    total: number;
  };
}

  'https://api.trusset.org/api/v2/custody/api/transactions?customerRef=CUST-12345',

);
