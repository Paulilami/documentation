---
title: 'Submit Risk Data'
api: 'POST /api/v2/risk/assets/{assetAddress}'
---

Create or update risk data for a tokenized asset. If a record exists, it will be updated and the previous values stored in history.

## Path Parameters

<ParamField path="assetAddress" type="string" required>
  The token contract address
</ParamField>

## Body Parameters

<ParamField body="riskScore" type="number" required>
  Risk score from 1 (lowest risk) to 100 (highest risk)
</ParamField>

<ParamField body="riskGrade" type="string">
  Optional letter grade (A-F). Auto-calculated from score if not provided.
</ParamField>

<ParamField body="category" type="string" default="market">
  Risk category: `credit`, `market`, `liquidity`, `operational`, `regulatory`, `counterparty`
</ParamField>

<ParamField body="metadata" type="object">
  Additional issuer-specific risk details
</ParamField>

<ParamField body="reason" type="string">
  Reason for the update (stored in history)
</ParamField>

## Response Fields

<ResponseField name="id" type="string">Unique record identifier</ResponseField>
<ResponseField name="assetAddress" type="string">Token contract address</ResponseField>
<ResponseField name="riskScore" type="number">Current risk score</ResponseField>
<ResponseField name="riskGrade" type="string">Current letter grade</ResponseField>
<ResponseField name="category" type="string">Risk category</ResponseField>
<ResponseField name="metadata" type="object">Additional details</ResponseField>
<ResponseField name="updatedAt" type="string">Last update timestamp</ResponseField>

  metadata: { requestId: string; timestamp: string };
