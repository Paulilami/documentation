---
title: 'Get Risk History'
api: 'GET /api/v2/risk/assets/{assetAddress}/history'
---

Retrieve the complete history of risk updates for an asset, providing a full audit trail.

## Path Parameters

<ParamField path="assetAddress" type="string" required>
  The token contract address
</ParamField>

## Query Parameters

<ParamField query="limit" type="number" default="50">
  Maximum history entries to return (max 100)
</ParamField>

## Response Fields

<ResponseField name="assetAddress" type="string">Token contract address</ResponseField>
<ResponseField name="currentScore" type="number">Current risk score</ResponseField>
<ResponseField name="currentGrade" type="string">Current letter grade</ResponseField>
<ResponseField name="history" type="array">Array of historical updates</ResponseField>

### History Entry Fields

<ResponseField name="previousScore" type="number">Score before this update</ResponseField>
<ResponseField name="previousGrade" type="string">Grade before this update</ResponseField>
<ResponseField name="newScore" type="number">Score after this update</ResponseField>
<ResponseField name="newGrade" type="string">Grade after this update</ResponseField>
<ResponseField name="category" type="string">Risk category</ResponseField>
<ResponseField name="reason" type="string">Reason for update</ResponseField>
<ResponseField name="createdAt" type="string">Timestamp of update</ResponseField>


```javascript JavaScript
const response = await fetch(
  'https://api.trusset.org/api/v2/risk/assets/0xAbC123dEf456789012345678901234567890AbCd/history?limit=10',
  {
    headers: {
      'x-api-key': 'trusset_v2_abc123xy_a1b2c3d4e5f6...'
    }
  }
);

const data = await response.json();
data.data.history.forEach(entry => {
  console.log(`${entry.createdAt}: ${entry.previousGrade} → ${entry.newGrade}`);
});
```
```python Python
import requests

response = requests.get(
    'https://api.trusset.org/api/v2/risk/assets/0xAbC123dEf456789012345678901234567890AbCd/history',
    headers={
        'x-api-key': 'trusset_v2_abc123xy_a1b2c3d4e5f6...'
    },
    params={'limit': 10}
)

data = response.json()
for entry in data['data']['history']:
    print(f"{entry['createdAt']}: {entry['previousGrade']} → {entry['newGrade']}")
```
```typescript TypeScript
interface HistoryEntry {
  id: string;
  previousScore: number | null;
  previousGrade: string | null;
  newScore: number;
  newGrade: string;
  category: string;
  reason: string | null;
  createdAt: string;
}

interface HistoryResponse {
  success: boolean;
  data: {
    assetAddress: string;
    currentScore: number;
    currentGrade: string;
    history: HistoryEntry[];
  };
}

const response = await fetch(
  'https://api.trusset.org/api/v2/risk/assets/0xAbC123.../history',
  { headers: { 'x-api-key': 'trusset_v2_abc123xy_...' } }
);
const data: HistoryResponse = await response.json();
```
