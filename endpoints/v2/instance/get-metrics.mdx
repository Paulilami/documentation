---
title: 'Get Instance Metrics'
api: 'GET /api/v2/instance/metrics'
---

Retrieve usage metrics and recent activity for the current instance.

## Response Fields

<ResponseField name="counts" type="object">
  <Expandable title="properties">
    <ResponseField name="tokens" type="number">Total deployed tokens</ResponseField>
    <ResponseField name="customers" type="number">Total active customers</ResponseField>
    <ResponseField name="webhooks" type="number">Active webhook configurations</ResponseField>
  </Expandable>
</ResponseField>

<ResponseField name="recentActivity" type="array">
  List of recent actions performed on the instance
  <Expandable title="item properties">
    <ResponseField name="action" type="string">Action type identifier</ResponseField>
    <ResponseField name="timestamp" type="string">ISO 8601 timestamp</ResponseField>
  </Expandable>
</ResponseField>

<RequestExample>
```bash cURL
curl -X GET "https://api.trusset.org/api/v2/instance/metrics" \
  -H "x-api-key: trusset_v2_abc123xy_a1b2c3d4e5f6..."
```
```javascript JavaScript
const response = await fetch('https://api.trusset.org/api/v2/instance/metrics', {
  method: 'GET',
  headers: {
    'x-api-key': 'trusset_v2_abc123xy_a1b2c3d4e5f6...'
  }
});

const data = await response.json();
console.log(`Total tokens: ${data.data.counts.tokens}`);
console.log(`Total customers: ${data.data.counts.customers}`);
```
```python Python
import requests

response = requests.get(
    'https://api.trusset.org/api/v2/instance/metrics',
    headers={
        'x-api-key': 'trusset_v2_abc123xy_a1b2c3d4e5f6...'
    }
)

data = response.json()
print(f"Total tokens: {data['data']['counts']['tokens']}")
```
```typescript TypeScript
interface MetricsResponse {
  success: boolean;
  data: {
    counts: {
      tokens: number;
      customers: number;
      webhooks: number;
    };
    recentActivity: Array<{
      action: string;
      timestamp: string;
    }>;
  };
  metadata: {
    requestId: string;
    timestamp: string;
  };
}

const response = await fetch('https://api.trusset.org/api/v2/instance/metrics', {
  headers: { 'x-api-key': 'trusset_v2_abc123xy_...' }
});
const data: MetricsResponse = await response.json();
```
