---
title: 'List API Keys'
api: 'GET /api/v2/instance/api-keys'
---

Retrieve all API keys associated with the current instance. Keys are returned with a preview only (first 20 characters) for security.

## Response Fields

<ResponseField name="data" type="array">
  List of API keys
  <Expandable title="item properties">
    <ResponseField name="id" type="string">Unique key identifier</ResponseField>
    <ResponseField name="name" type="string">Display name for the key</ResponseField>
    <ResponseField name="keyPreview" type="string">First 20 characters of the key</ResponseField>
    <ResponseField name="version" type="number">API key version (1 or 2)</ResponseField>
    <ResponseField name="status" type="string">`ACTIVE`, `REVOKED`, `EXPIRED`, or `PENDING_ROTATION`</ResponseField>
    <ResponseField name="lastUsedAt" type="string">Last usage timestamp (nullable)</ResponseField>
    <ResponseField name="createdAt" type="string">Creation timestamp</ResponseField>
  </Expandable>
</ResponseField>

<RequestExample>
```bash cURL
curl -X GET "https://api.trusset.org/api/v2/instance/api-keys" \
  -H "x-api-key: trusset_v2_abc123xy_a1b2c3d4e5f6..."
```
```javascript JavaScript
const response = await fetch('https://api.trusset.org/api/v2/instance/api-keys', {
  method: 'GET',
  headers: {
    'x-api-key': 'trusset_v2_abc123xy_a1b2c3d4e5f6...'
  }
});

const data = await response.json();
data.data.forEach(key => {
  console.log(`${key.name}: ${key.status}`);
});
```
```python Python
import requests

response = requests.get(
    'https://api.trusset.org/api/v2/instance/api-keys',
    headers={
        'x-api-key': 'trusset_v2_abc123xy_a1b2c3d4e5f6...'
    }
)

data = response.json()
for key in data['data']:
    print(f"{key['name']}: {key['status']}")
```
```typescript TypeScript
interface ApiKey {
  id: string;
  name: string;
  keyPreview: string;
  version: number;
  status: 'ACTIVE' | 'REVOKED' | 'EXPIRED' | 'PENDING_ROTATION';
  lastUsedAt: string | null;
  createdAt: string;
}

interface ApiKeysResponse {
  success: boolean;
  data: ApiKey[];
  metadata: {
    requestId: string;
    timestamp: string;
  };
}

const response = await fetch('https://api.trusset.org/api/v2/instance/api-keys', {
  headers: { 'x-api-key': 'trusset_v2_abc123xy_...' }
});
const data: ApiKeysResponse = await response.json();
```
