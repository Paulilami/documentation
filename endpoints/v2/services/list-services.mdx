---
title: 'List Services'
api: 'GET /api/v2/services'
---

Retrieve all services enabled for the current instance, including their capabilities and available endpoints.

## Response Fields

<ResponseField name="services" type="array">
  List of enabled services
  <Expandable title="item properties">
    <ResponseField name="id" type="string">Service identifier</ResponseField>
    <ResponseField name="name" type="string">Human-readable service name</ResponseField>
    <ResponseField name="enabled" type="boolean">Whether service is active</ResponseField>
    <ResponseField name="endpoints" type="array">List of available API endpoints</ResponseField>
    <ResponseField name="stats" type="object">Service-specific usage statistics</ResponseField>
  </Expandable>
</ResponseField>

<ResponseField name="network" type="object">
  Network configuration for the instance
  <Expandable title="properties">
    <ResponseField name="type" type="string">Network type</ResponseField>
    <ResponseField name="isTestnet" type="boolean">Whether using testnet</ResponseField>
    <ResponseField name="chainId" type="number|string">Chain identifier</ResponseField>
  </Expandable>
</ResponseField>

## Available Services

| Service ID | Name | Description |
|------------|------|-------------|
| `tokenization` | Tokenization | Deploy and manage tokenized assets |
| `oracle` | Price Oracles | Real-time price feeds and data verification |
| `stablecoin` | Stablecoins | Deploy and manage stablecoins |
| `gold` | Gold Tokens | Commodity-backed token management |
| `kyc` | KYC Management | Customer identity verification |
| `customers` | Customer Management | Manage customer records |

<RequestExample>
```bash cURL
curl -X GET "https://api.trusset.org/api/v2/services" \
  -H "x-api-key: trusset_v2_abc123xy_a1b2c3d4e5f6..."
```
```javascript JavaScript
const response = await fetch('https://api.trusset.org/api/v2/services', {
  method: 'GET',
  headers: {
    'x-api-key': 'trusset_v2_abc123xy_a1b2c3d4e5f6...'
  }
});

const data = await response.json();
data.data.services.forEach(service => {
  console.log(`${service.name}: ${service.enabled ? 'Enabled' : 'Disabled'}`);
  console.log(`  Endpoints: ${service.endpoints.join(', ')}`);
});
```
```python Python
import requests

response = requests.get(
    'https://api.trusset.org/api/v2/services',
    headers={
        'x-api-key': 'trusset_v2_abc123xy_a1b2c3d4e5f6...'
    }
)

data = response.json()
for service in data['data']['services']:
    status = 'Enabled' if service['enabled'] else 'Disabled'
    print(f"{service['name']}: {status}")
    print(f"  Endpoints: {', '.join(service['endpoints'])}")
```
```typescript TypeScript
interface Service {
  id: string;
  name: string;
  enabled: boolean;
  endpoints: string[];
  stats?: Record<string, number>;
}

interface ServicesResponse {
  success: boolean;
  data: {
    services: Service[];
    network: {
      type: string;
      isTestnet: boolean;
      chainId: number | string;
    };
  };
  metadata: {
    requestId: string;
    timestamp: string;
  };
}

const response = await fetch('https://api.trusset.org/api/v2/services', {
  headers: { 'x-api-key': 'trusset_v2_abc123xy_...' }
});
const data: ServicesResponse = await response.json();
```
