---
title: 'Get All Documents'
api: 'GET /api/v2/light-tokens/documents/{tokenAddress}'
---

Retrieve all compliance documents attached to a Light Token.

## Path Parameters

<ParamField path="tokenAddress" type="string" required>
  Address of the Light Token contract
</ParamField>

## Response Fields

<ResponseField name="documents" type="array">
  List of documents
  <Expandable title="document properties">
    <ResponseField name="name" type="string">Document identifier</ResponseField>
    <ResponseField name="nameHash" type="string">Document name as bytes32</ResponseField>
    <ResponseField name="uri" type="string">Document URI</ResponseField>
    <ResponseField name="documentHash" type="string">Content hash for verification</ResponseField>
    <ResponseField name="lastModified" type="number">Unix timestamp of last modification</ResponseField>
  </Expandable>
</ResponseField>

```javascript JavaScript
const response = await fetch(
  'https://api.trusset.org/api/v2/light-tokens/documents/0xAbC123dEf456789012345678901234567890AbCd',
  {
    headers: {
      'x-api-key': 'trusset_v2_abc123xy_a1b2c3d4e5f6...'
    }
  }
);

const data = await response.json();
data.data.documents.forEach(doc => {
  console.log(`${doc.name}: ${doc.uri}`);
});
```
```python Python
import requests

response = requests.get(
    'https://api.trusset.org/api/v2/light-tokens/documents/0xAbC123dEf456789012345678901234567890AbCd',
    headers={
        'x-api-key': 'trusset_v2_abc123xy_a1b2c3d4e5f6...'
    }
)

data = response.json()
for doc in data['data']['documents']:
    print(f"{doc['name']}: {doc['uri']}")
```
```typescript TypeScript
interface Document {

  nameHash: string;
  uri: string;
  documentHash: string;
  lastModified: number;
}

interface DocumentsResponse {
  success: boolean;
  data: {
    documents: Document[];
  };
  metadata: {
    requestId: string;
    timestamp: string;
  };
}

  'https://api.trusset.org/api/v2/light-tokens/documents/0xAbC123dEf456789012345678901234567890AbCd',
  { headers: { 'x-api-key': 'trusset_v2_abc123xy_...' } }
);
const data: DocumentsResponse = await response.json();
```
