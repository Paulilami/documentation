---
title: 'Get Instance Tokens'
api: 'GET /api/v2/light-tokens/instance'
---

Retrieve all Light Tokens associated with the current instance. This is scoped to tokens deployed through or registered with your Trusset instance.

## Response Fields

<ResponseField name="tokens" type="array">
  List of tokens in the current instance
  <Expandable title="token properties">
    <ResponseField name="address" type="string">Token contract address</ResponseField>
    <ResponseField name="name" type="string">Token name</ResponseField>
    <ResponseField name="symbol" type="string">Token symbol</ResponseField>
    <ResponseField name="decimals" type="number">Number of decimals</ResponseField>
    <ResponseField name="totalSupply" type="string">Total supply</ResponseField>
    <ResponseField name="ivc" type="string">Governing IVC address</ResponseField>
    <ResponseField name="createdAt" type="string">Creation timestamp</ResponseField>
  </Expandable>
</ResponseField>
<ResponseField name="count" type="number">Total number of tokens in instance</ResponseField>

<RequestExample>
```bash cURL
curl -X GET "https://api.trusset.org/api/v2/light-tokens/instance" \
  -H "x-api-key: trusset_v2_abc123xy_a1b2c3d4e5f6..."
```
```javascript JavaScript
const response = await fetch(
  'https://api.trusset.org/api/v2/light-tokens/instance',
  {
    headers: {
      'x-api-key': 'trusset_v2_abc123xy_a1b2c3d4e5f6...'
    }
  }
);

const data = await response.json();
data.data.tokens.forEach(token => {
  console.log(`${token.name} (${token.symbol}): ${token.address}`);
});
```
```python Python
import requests

response = requests.get(
    'https://api.trusset.org/api/v2/light-tokens/instance',
    headers={
        'x-api-key': 'trusset_v2_abc123xy_a1b2c3d4e5f6...'
    }
)

data = response.json()
for token in data['data']['tokens']:
    print(f"{token['name']} ({token['symbol']}): {token['address']}")
```
```typescript TypeScript
interface InstanceToken {
  address: string;
  name: string;
  symbol: string;
  decimals: number;
  totalSupply: string;
  ivc: string;
  createdAt: string;
}

interface InstanceTokensResponse {
  success: boolean;
  data: {
    tokens: InstanceToken[];
    count: number;
  };
  metadata: {
    requestId: string;
    timestamp: string;
  };
}

const response = await fetch(
  'https://api.trusset.org/api/v2/light-tokens/instance',
  { headers: { 'x-api-key': 'trusset_v2_abc123xy_...' } }
);
const data: InstanceTokensResponse = await response.json();
```
