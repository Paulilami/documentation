---
title: 'Get Token Holders'
api: 'GET /api/v2/light-tokens/holders/{tokenAddress}'
---

Retrieve a list of all token holders for a Light Token. Results are retrieved from the database for the current instance.

## Path Parameters

<ParamField path="tokenAddress" type="string" required>
  Address of the Light Token contract
</ParamField>

## Response Fields

<ResponseField name="holders" type="array">
  List of token holders
  <Expandable title="holder properties">
    <ResponseField name="address" type="string">Holder's wallet address</ResponseField>
    <ResponseField name="balance" type="string">Total token balance</ResponseField>
    <ResponseField name="frozen" type="boolean">Whether the account is frozen</ResponseField>
  </Expandable>
</ResponseField>
<ResponseField name="totalHolders" type="number">Total number of unique holders</ResponseField>

<RequestExample>
```bash cURL
curl -X GET "https://api.trusset.org/api/v2/light-tokens/holders/0xAbC123dEf456789012345678901234567890AbCd" \
  -H "x-api-key: trusset_v2_abc123xy_a1b2c3d4e5f6..."
```
```javascript JavaScript
const response = await fetch(
  'https://api.trusset.org/api/v2/light-tokens/holders/0xAbC123dEf456789012345678901234567890AbCd',
  {
    headers: {
      'x-api-key': 'trusset_v2_abc123xy_a1b2c3d4e5f6...'
    }
  }
);

const data = await response.json();
console.log('Total holders:', data.data.totalHolders);
data.data.holders.forEach(h => console.log(`${h.address}: ${h.balance}`));
```
```python Python
import requests

response = requests.get(
    'https://api.trusset.org/api/v2/light-tokens/holders/0xAbC123dEf456789012345678901234567890AbCd',
    headers={
        'x-api-key': 'trusset_v2_abc123xy_a1b2c3d4e5f6...'
    }
)

data = response.json()
print(f"Total holders: {data['data']['totalHolders']}")
for holder in data['data']['holders']:
    print(f"{holder['address']}: {holder['balance']}")
```

  { headers: { 'x-api-key': 'trusset_v2_abc123xy_...' } }

const data: HoldersResponse = await response.json();
