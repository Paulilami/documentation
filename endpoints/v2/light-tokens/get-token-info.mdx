---
title: 'Get Token Info'
api: 'GET /api/v2/light-tokens/info/{tokenAddress}'
---

Retrieve comprehensive information about a Light Token including its configuration, total supply, and governance details.

## Path Parameters

<ParamField path="tokenAddress" type="string" required>
  Address of the Light Token contract
</ParamField>

## Response Fields

<ResponseField name="address" type="string">Token contract address</ResponseField>
<ResponseField name="name" type="string">Token name</ResponseField>
<ResponseField name="symbol" type="string">Token symbol</ResponseField>
<ResponseField name="decimals" type="number">Number of decimal places</ResponseField>
<ResponseField name="totalSupply" type="string">Total token supply</ResponseField>
<ResponseField name="ivc" type="string">Instance Verification Contract address</ResponseField>
<ResponseField name="metadataURI" type="string">Metadata URI</ResponseField>
<ResponseField name="transfersRestricted" type="boolean">Whether transfers are globally restricted</ResponseField>
<ResponseField name="identityRegistry" type="string">Associated identity registry address</ResponseField>

```typescript TypeScript
interface TokenInfo {
  address: string;
  name: string;
  symbol: string;
  decimals: number;
  totalSupply: string;
  ivc: string;
  metadataURI: string;
  transfersRestricted: boolean;
  identityRegistry: string;
}

interface TokenInfoResponse {
  success: boolean;
  data: TokenInfo;
  metadata: {
    requestId: string;
    timestamp: string;
  };
}

const response = await fetch(
  'https://api.trusset.org/api/v2/light-tokens/info/0xAbC123dEf456789012345678901234567890AbCd',
  { headers: { 'x-api-key': 'trusset_v2_abc123xy_...' } }
);
const data: TokenInfoResponse = await response.json();
```
